```python
import os
import sys
import datetime
import sys

path = sys.argv[1]
target = sys.argv[2]

# date_str = '2023-02-28 14:30:00'
date_format = '%Y-%m-%d %H:%M:%S'
date_obj = datetime.datetime.strptime(target, date_format)
print( os.stat(path) );
timestamp = int(date_obj.timestamp())
atime = timestamp
mtime = timestamp
tup = (atime, mtime)
os.utime(path, tup)
print( os.stat(path) );
```
